Terminology

Term Description
True Positive Benchmark actually has the bug, and the tool reports that the benchmark has the bug.
True Negative Benchmark does not actually have the bug, and the tool does not report that the benchmark has the bug.
False Positive Benchmark does not actually have the bug, but the tool reports that the benchmark has the bug.
False Negative Benchmark actually has the bug, but the tool does not report that the benchmark has the bug.
False Negative Benchmark actually has the bug, but the tool does not report that the benchmark has the bug.
Unsupported The tool does not support this particular bug type.
Analysis Failed The tool prints an exception.
Timed out The tool did not terminate within the specified time limit.

//integer_overflow_benign_2.sol
//Single transaction overflow
//Post-transaction effect: overflow nevers escapes to publicly-readable storage

pragma solidity ^0.4.19;

contract Benchmark {
    uint public count;

    function run(uint256 input) public {
        bool err;
        uint256 res = 1;

        (err, res) = minus(res, input);

        require(!err);
        count = res;
    }

    //from BasicMathLib
    function minus(uint256 a, uint256 b) private pure returns (bool err,uint256 res) {
        assembly {
            res := sub(a,b)
            switch eq(and(eq(add(res,b), a), or(lt(res,a), eq(res,a))), 1)
            case 0 {
                err := 1
                res := 0
            }
        }
    }
}

//eth_tx_order_dependence_puzzle.sol
//Adapted from Oyente paper
pragma solidity ^0.4.16;

contract Benchmark {
    address public owner ;
    bool public locked ;
    uint public reward ;
    bytes32 public diff ;
    bytes public solution;

    function Puzzle() public payable {
        owner = msg.sender;
        reward = msg.value;
        locked = false ;
        diff = bytes32 (11111); // pre - defined difficulty
    }

    function () public payable { // main code , runs at every invocation
        if ( msg.sender == owner ){ // update reward
            require (!locked);
            owner.transfer(reward);
            reward = msg.value;
        } else {
            if (msg.data.length > 0) { // submit a solution
                require (!locked);
                if ( sha256 (msg.data ) < diff ){
                    msg.sender.transfer(reward); // send reward
                    solution = msg.data;
                    locked = true;
                }
            }
        }
    }
}
